version: '3.8'

services:
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: damul-backend
    ports:
      - "3000:3000"
    environment:
      - SPRING_PROFILES_ACTIVE=prod
    restart: always
    networks:
      - damul-network

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: damul-frontend
    ports:
      - "5173:5173"
    depends_on:
      - backend
    restart: always
    networks:
      - damul-network

  mysql:
    image: mysql:8.4.5
    container_name: damul-mysql
    environment:
      MYSQL_ROOT_PASSWORD: ssafy
      MYSQL_DATABASE: damul
      MYSQL_USER: damul
      MYSQL_PASSWORD: ssafy
    ports:
      - "3306:3306"
    volumes:
      - mysql-data:/var/lib/mysql
    networks:
      - damul-network
    restart: always

redis:
    image: redis:latest
    container_name: damul-redis
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    networks:
      - damul-network
    restart: always

nginx:
    image: nginx:alpine
    container_name: damul-nginx
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/conf.d/default.conf
    depends_on:
      - frontend
      - backend
    networks:
      - damul-network
    restart: always



networks:
  damul-network:
    driver: bridge

volumes:
  mysql-data:
  redis-data:
